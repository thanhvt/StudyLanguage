### 5.3 Review Scheduling

```typescript
// Pseudo-code for scheduling
function scheduleNextReview(word: Word, remembered: boolean) {
  if (remembered) {
    word.level = Math.min(word.level + 1, 5);
  } else {
    word.level = 1; // Reset to beginning
  }
  
  const intervals = [0, 1, 3, 7, 30]; // days
  word.nextReview = addDays(now(), intervals[word.level - 1]);
  
  return word;
}
```
